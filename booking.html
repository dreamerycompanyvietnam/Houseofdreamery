<script>

let currentStep = 0;

const steps = document.querySelectorAll(".step");

const circles = document.querySelectorAll(".circle");


function updateUI(){

  steps.forEach(s=>s.classList.remove("active"));

  steps[currentStep].classList.add("active");


  circles.forEach((c,i)=>{

    c.classList.toggle("active", i<=currentStep);

  });

}


document.querySelectorAll(".next").forEach(btn=>{

  btn.onclick = ()=>{

    currentStep++;


    if(currentStep===4){

      confirm.innerHTML = `

👤 ${name.value}<br>

📞 ${phone.value}<br>

✂️ ${service.value}<br>

💈 ${barber.value} (${barberPrice.value})<br>

📅 ${date.value} ${time.value}

      `;

    }


    updateUI();

  }

});


function selectService(el){

  document.querySelectorAll(".card").forEach(c=>c.classList.remove("active"));

  el.classList.add("active");

  service.value = el.innerText;

}


function selectBarber(el,name,price){

  document.querySelectorAll(".barber-card").forEach(b=>b.classList.remove("active"));

  el.classList.add("active");

  barber.value = name;

  barberPrice.value = price;

}


function selectTime(el){

  document.querySelectorAll(".time").forEach(t=>t.classList.remove("active"));

  el.classList.add("active");

  time.value = el.innerText;

}


submitBtn.onclick = ()=>{

  msg.innerText = "ĐẶT LỊCH THÀNH CÔNG!";

};

</script>
